class ChaptersModel:

        from PySide6.QtCore import QSettings
        self.s = QSettings('DeepParser','–ü–µ—Ä–µ–≤–æ–¥—á–∏–∫')

        self.out_dir = Path(self.s.value('paths/out','') or '.')

    def names(self):
        return [p.name for p in self.files]

    def count(self):
        return len(self.files)

    def load_text(self, idx:int) -> str:
        p = self.files[idx]
        if p.suffix.lower()=='.txt':
            return p.read_text(encoding='utf-8', errors='ignore')
        else:
            d = Document(str(p))
            return '\n'.join([p.text for p in d.paragraphs])

    def save_translation(self, idx:int, text:str):
        src = self.files[idx]
        name = src.stem + '.docx'
        self.out_dir.mkdir(parents=True, exist_ok=True)
        dst = self.out_dir / name
        d = Document();
        for line in text.split('\n'):
            d.add_paragraph(line)
        d.save(str(dst))
